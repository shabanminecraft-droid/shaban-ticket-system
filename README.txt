let token=localStorage.getItem('shaban_admin_token')||'';let tickets=[];let selected=null;function qs(id){return document.getElementById(id)}function headers(){return {'Content-Type':'application/json','Authorization':'Bearer '+token}}function err(msg){qs('err').textContent=msg;qs('err').style.display=msg?'block':'none'}
async function adminLogin(){const r=await fetch('/api/admin/login',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({password:qs('pw').value})});const data=await r.json();if(!r.ok)return err(data.error||'Wrong password');token=data.token;localStorage.setItem('shaban_admin_token',token);boot()}
